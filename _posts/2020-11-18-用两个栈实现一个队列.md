#  用两个栈来实现一个队列，完成队列的Push和Pop操作。 队列中的元素为int类型。

```python
# -*- coding:utf-8 -*-
class Solution:
    def __init__(self):
        self.a_stack = []
        self.b_stack = []
    def push(self, node):
        self.a_stack.append(node)
        
    def pop(self):
        if self.b_stack:
            return self.b_stack.pop()
        if not self.a_stack:
            return None
        else:
            while self.a_stack:
                self.b_stack.append(self.a_stack.pop())
            return self.b_stack.pop()
                
```
